    <script>

        let videoFile = null;

        let videoDuration = 0;

        let ffmpeg = null;

        let currentMode = 'summary';

        

        const config = {

            segmentDuration: 0.5,

            totalDuration: 15,

            quality: 'medium',

            fps: 30,

            width: 640,

            clipStart: 0

        };


        document.addEventListener('DOMContentLoaded', () => {

            initDropZone();

            loadFFmpeg();

        async function loadFFmpeg() {

            try {

                const { FFmpeg } = FFmpegWASM;

                ffmpeg = new FFmpeg();

                

                ffmpeg.on('progress', ({ progress }) => {

                    const p = Math.round(progress * 100);

                    updateProgress(p);

                    document.getElementById('processDetail').textContent = 

                        currentMode === 'summary' ? `Procesando ${Math.ceil(config.totalDuration/config.segmentDuration)} segmentos...` : 'Generando GIF...';

                });


                await ffmpeg.load();

                updateStatus('Sistema listo');

            } catch (e) {

                console.error(e);

                showToast('Error al cargar el procesador');

            }

        }


        function initDropZone() {

            const zone = document.getElementById('dropZone');

            

            ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(e => {

                zone.addEventListener(e, (ev) => { ev.preventDefault(); ev.stopPropagation(); }, false);

            });

            

            ['dragenter', 'dragover'].forEach(e => {

                zone.addEventListener(e, () => zone.classList.add('drag-over'), false);

            });

            

            ['dragleave', 'drop'].forEach(e => {

                zone.addEventListener(e, () => zone.classList.remove('drag-over'), false);

            });

            

            zone.addEventListener('drop', (e) => handleFiles(e.dataTransfer.files), false);

        }


        function handleFileSelect(e) { handleFiles(e.target.files); }


        function handleFiles(files) {

            if (files.length) {

                const file = files[0];

                if (!file.type.startsWith('video/')) return showToast('Selecciona un archivo de video');

                

                videoFile = file;

                loadVideo(file);

            }

        }


        async function loadVideo(file) {

            const url = URL.createObjectURL(file);

            const video = document.createElement('video');

            video.src = url;

            

            video.onloadedmetadata = () => {

                videoDuration = video.duration;

                

                document.getElementById('videoName').textContent = file.name;

                document.getElementById('videoMeta').textContent = 

                    `Duración: ${formatTime(videoDuration)} • ${(file.size/1024/1024).toFixed(1)} MB`;

                

                document.getElementById('uploadSection').classList.add('hidden');

                document.getElementById('editorSection').classList.remove('hidden');

                

                updateSummaryConfig();

                

                // Setup clip mode max

                const clipStart = document.getElementById('clipStart');

                clipStart.max = Math.max(0, videoDuration - 15);

                updateClipConfig();

            };

        }


        function setMode(mode) {

            currentMode = mode;

            document.querySelectorAll('.mode-btn').forEach(b => b.classList.remove('active'));

            document.getElementById(`mode-${mode}`).classList.add('active');

            

            if (mode === 'summary') {

                document.getElementById('summaryControls').classList.remove('hidden');

                document.getElementById('clipControls').classList.add('hidden');

                document.getElementById('btnText').textContent = 'Crear Resumen GIF';

            } else {

                document.getElementById('summaryControls').classList.add('hidden');

                document.getElementById('clipControls').classList.remove('hidden');

                document.getElementById('btnText').textContent = 'Crear Clip GIF';

            }

        }


        function updateSummaryConfig() {

            config.segmentDuration = parseFloat(document.getElementById('segmentDuration').value);

            config.totalDuration = parseInt(document.getElementById('totalDuration').value);

            

            const segments = Math.ceil(config.totalDuration / config.segmentDuration);

            

            document.getElementById('segmentDurationDisplay').textContent = config.segmentDuration + 's';

            document.getElementById('totalDurationDisplay').textContent = config.totalDuration + 's';

            document.getElementById('totalSegments').textContent = segments;

            

            // Update chips

            document.getElementById('segmentInfo').innerHTML = `

                <div class="chip">${segments} pedacitos</div>

                <div class="chip">${config.segmentDuration}s c/u</div>

                <div class="chip">${config.totalDuration}s total</div>

            `;

            

            // Draw visualizer

            drawSegmentVisualizer(segments);

        }


        function drawSegmentVisualizer(totalSegments) {

            const container = document.getElementById('segmentVisualizer');

            container.innerHTML = '';

            

            for (let i = 0; i < totalSegments; i++) {

                const bar = document.createElement('div');

                bar.className = 'segment-bar';

                

                const left = (i / totalSegments) * 100;

                const width = (0.8 / totalSegments) * 100; // 80% of slot width

                

                bar.style.left = left + '%';

                bar.style.width = width + '%';

                bar.style.opacity = 0.6 + (i % 3) * 0.2; // Varied opacity

                

                container.appendChild(bar);

            }

        }


        function updateClipConfig() {

            const start = parseInt(document.getElementById('clipStart').value);

            const end = Math.min(start + 15, videoDuration);

            

            config.clipStart = start;

            

            document.getElementById('clipStartTime').textContent = formatTime(start);

            document.getElementById('clipEndTime').textContent = formatTime(end);

            document.getElementById('clipDurationDisplay').textContent = (end - start).toFixed(1) + 's';

        }


        async function startConversion() {

            if (!ffmpeg || !videoFile) return showToast('Espera la carga...');

            

            config.quality = document.getElementById('quality').value;

            config.fps = document.getElementById('fps').value;

            config.width = document.getElementById('resolution').value;

            

            document.getElementById('editorSection').classList.add('hidden');

            document.getElementById('processingSection').classList.remove('hidden');

            

            try {

                const { fetchFile } = FFmpegUtil;

                const inputName = 'input.mp4';

                const outputName = 'output.gif';

                

                await ffmpeg.writeFile(inputName, await fetchFile(videoFile));

                

                let command;

                

                if (currentMode === 'summary') {

                    await createSummaryGIF(inputName, outputName);

                } else {

                    await createClipGIF(inputName, outputName);

                }

                

                const data = await ffmpeg.readFile(outputName);

                gifBlob = new Blob([data.buffer], { type: 'image/gif' });

                

                await ffmpeg.deleteFile(inputName);

                await ffmpeg.deleteFile(outputName);

                

                showResult();

                

            } catch (error) {

                console.error(error);

                showToast('Error: ' + error.message);

                document.getElementById('processingSection').classList.add('hidden');

                document.getElementById('editorSection').classList.remove('hidden');

            }

        }


        async function createSummaryGIF(input, output) {

            const segmentCount = Math.ceil(config.totalDuration / config.segmentDuration);

            const interval = videoDuration / segmentCount;

            

            updateProcessStatus('Generando resumen...', 10);

            

            // Build complex filter for concatenating segments

            let filterComplex = '';

            let inputs = [];

            

            for (let i = 0; i < segmentCount; i++) {

                const start = i * interval;

                const end = start + config.segmentDuration;

                

                // For each segment: trim, scale, setpts

                filterComplex += `[0:v]trim=start=${start}:end=${end},setpts=PTS-STARTPTS,scale=w=${config.width}:h=-1:flags=lanczos,fps=${config.fps}[v${i}];`;

                inputs.push(`[v${i}]`);

            }

            

            // Concatenate all

            filterComplex += inputs.join('') + `concat=n=${segmentCount}:v=1:a=0[outv]`;

            

            // Palettegen for better colors

            const paletteFilter = "split[s0][s1];[s0]palettegen=max_colors=128[p];[s1][p]paletteuse";

            

            await ffmpeg.exec([

                '-i', input,

                '-filter_complex', filterComplex + ';[outv]' + paletteFilter,

                '-loop', '0',

                output

            ]);

        }


        async function createClipGIF(input, output) {

            const filters = [

                `fps=${config.fps}`,

                `scale=w=${config.width}:h=-1:flags=lanczos`,

                `split[s0][s1];[s0]palettegen=max_colors=${config.quality === 'high' ? 256 : 128}[p];[s1][p]paletteuse`

            ].join(',');

            

            await ffmpeg.exec([

                '-ss', config.clipStart.toString(),

                '-t', '15',

                '-i', input,

                '-vf', filters,

                '-loop', '0',

                output

            ]);

        }


        function updateProcessStatus(title, progress) {

            document.getElementById('processTitle').textContent = title;

            updateProgress(progress);

        }


        function updateProgress(percent) {

            const circle = document.getElementById('progressCircle');

            const r = circle.r.baseVal.value;

            const c = r * 2 * Math.PI;

            const offset = c - (percent / 100) * c;

            

            circle.style.strokeDashoffset = offset;

            document.getElementById('progressText').textContent = percent + '%';

            document.getElementById('progressBar').style.width = percent + '%';

        }


        function showResult() {

            document.getElementById('processingSection').classList.add('hidden');

            document.getElementById('resultSection').classList.remove('hidden');

            

            const url = URL.createObjectURL(gifBlob);

            document.getElementById('resultImage').src = url;

            

            const size = (gifBlob.size / 1024).toFixed(1);

            const segments = currentMode === 'summary' ? 

                `${Math.ceil(config.totalDuration/config.segmentDuration)} segmentos` : 'Clip continuo';

            

            document.getElementById('fileInfo').textContent = 

                `GIF • ${size} KB • ${segments} • ${config.totalDuration || 15}s`;

            

            showToast('¡GIF creado exitosamente!');

        }


        function downloadGIF() {

            if (!gifBlob) return;

            const url = URL.createObjectURL(gifBlob);

            const a = document.createElement('a');

            a.href = url;

            a.download = `resumen-${Date.now()}.gif`;

            a.click();

            URL.revokeObjectURL(url);

            showToast('Descargando...');

        }


        function resetApp() {

            videoFile = null;

            gifBlob = null;

            videoDuration = 0;

            document.getElementById('fileInput').value = '';

            document.getElementById('uploadSection').classList.remove('hidden');

            document.getElementById('editorSection').classList.add('hidden');

            document.getElementById('processingSection').classList.add('hidden');

            document.getElementById('resultSection').classList.add('hidden');

            updateProgress(0);

        }


        function formatTime(s) {

            const m = Math.floor(s / 60);

            const sec = Math.floor(s % 60);

            return `${m.toString().padStart(2, '0')}:${sec.toString().padStart(2, '0')}`;

        }


        function showToast(msg) {

            const t = document.getElementById('toast');

            t.textContent = msg;

            t.classList.add('show');

            setTimeout(() => t.classList.remove('show'), 3000);

        }


        function updateStatus(txt) {

            document.getElementById('statusText').textContent = txt;

        }
